function toggleDarkMode() {
    document.body.classList.toggle("dark-mode");
    
    let p = document.getElementById("modeText");
    
    if (document.body.classList.contains("dark-mode")) {
        p.innerText = "Welcome!";
    } else {
        p.innerText = "Ambot dili nakooooooo";
    }
}
